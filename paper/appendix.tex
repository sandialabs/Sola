\section{Main nomenclature}

\textcolor{white}{Notation}
\begin{tabular}{rl}
    \textbf{PDE-constrained optimization} \\
    $\u \in \R^{n_u}$ & state variable \\
    $\z \in \R^{n_z}$ & control variable \\
    $J:\R^{n_u}\times\R^{n_z}\to\R$ & optimization objective \\
    $\c:\R^{n_u}\times\R^{n_z}\to\R^{n_u}$ & optimization constraint \\
    $\s:\R^{n_z}\to\R^{n_u}$ & constraint solution operator \\
    $\hat{J}:\R^{n_z}\to\R$ & reduced-space objective function
    \\[.0625in]

    \textbf{Hyper differential sensitivity analysis} \\
    $\tilde{\c}:\R^{n_u}\times\R^{n_z}\to\R^{n_u}$ & surrogate constraint \\
    $\tilde{\s}:\R^{n_z}\to\R^{n_u}$ & surrogate model \\
    $\tilde{J}:\R^{n_z}\to\R$ & surrogate model objective function \\
    $\tilde{\z}\in\R^{n_z}$ & solution to the surrogate-constrained optimization problem \\
    $\z'\in\R^{n_z}$ & solution to the original optimization problem \\
    $\bfdelta:\R^{n_z\times n_\theta}\to\R^{n_z}$ & model discrepancy term \\
    $\bftheta\in\R^{n_\theta}$ & parameters for the model discrepancy term \\
    $\J:\R^{n_z\times n_\theta}\to\R$ & parameterized objective with model discrepancy \\
    $\f:\R^{n_\theta}\to\R^{n_u}$ & implicit parameters-to-controls map \\
    $\M_u\in\R^{n_u\times n_u}$ & mass matrix for the state space \\
    $\M_z\in\R^{n_z\times n_z}$ & mass matrix for the control space \\
    $\W_u\in\R^{n_u\times n_u}$ & precision (inverse covariance) matrix for the state space \\
    $\W_z\in\R^{n_z\times n_z}$ & precision (inverse covariance) matrix for the control space \\
    % \\[.0625in]
\end{tabular}
