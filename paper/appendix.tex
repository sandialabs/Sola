\section{Main nomenclature}

\textcolor{white}{Notation}
\begin{tabular}{rl}
    \textbf{PDE-constrained optimization} \\
    $\u(t) \in \mathbb{R}^{n_u}$ & state variable \\
    $\z(t) \in \mathbb{R}^{n_z}$ & control variable \\
    $J:\R^{n_u}\times\R^{n_z}\to\R$ & optimization objective \\
    $\c:\R^{n_u}\times\R^{n_z}\to\R^{n_u}$ & optimization constraint \\
    $\s:\R^{n_z}\to\R^{n_u}$ & constraint solution operator \\
    $\hat{J}:\R^{n_z}\to\R$ & reduced-space objective function
    \\[.0625in]

    \textbf{Hyper differential sensitivity analysis} \\
    $\bfdelta:\R^{n_z\times n_\theta}\to\R^{n_z}$ & model discrepancy term \\
    $\bftheta\in\R^{n_\theta}$ & (unknown) parameters for the model discrepancy term \\
    $\J:\R^{n_z\times n_\theta}\to\R$ & objective with model discrepancy
\end{tabular}
